INSERT INTO member_relationship
(member_id_to, member_id_from, is_friend, created_at, updated_at)
SELECT
i.id,
you.id,
1,
NOW(),
NOW()
FROM
member i
JOIN
member you
ON
i.id <> you.id
LEFT OUTER JOIN
member_relationship relation
ON
relation.member_id_to = i.id
AND relation.member_id_from = you.id
WHERE
relation.id IS NULL
